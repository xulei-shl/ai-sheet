7#[cfg(test)]
8mod format_tests {
9
10    use crate::Color;
11
12    #[test]
13    fn test_hex_value() {
14        assert_eq!("FF000000", Color::Default.argb_hex_value());
15        assert_eq!("FF000000", Color::Black.argb_hex_value());
16        assert_eq!("FF0000FF", Color::Blue.argb_hex_value());
17        assert_eq!("FF800000", Color::Brown.argb_hex_value());
18        assert_eq!("FF00FFFF", Color::Cyan.argb_hex_value());
19        assert_eq!("FF808080", Color::Gray.argb_hex_value());
20        assert_eq!("FF008000", Color::Green.argb_hex_value());
21        assert_eq!("FF00FF00", Color::Lime.argb_hex_value());
22        assert_eq!("FFFF00FF", Color::Magenta.argb_hex_value());
23        assert_eq!("FF000080", Color::Navy.argb_hex_value());
24        assert_eq!("FFFF6600", Color::Orange.argb_hex_value());
25        assert_eq!("FFFFC0CB", Color::Pink.argb_hex_value());
26        assert_eq!("FF800080", Color::Purple.argb_hex_value());
27        assert_eq!("FFFF0000", Color::Red.argb_hex_value());
28        assert_eq!("FFC0C0C0", Color::Silver.argb_hex_value());
29        assert_eq!("FFFFFFFF", Color::White.argb_hex_value());
30        assert_eq!("FFFFFF00", Color::Yellow.argb_hex_value());
31        assert_eq!("FFABCDEF", Color::RGB(0xABCDEF).argb_hex_value());
32        assert_eq!("FF000000", Color::Theme(2, 1).argb_hex_value());
33    }
34}
