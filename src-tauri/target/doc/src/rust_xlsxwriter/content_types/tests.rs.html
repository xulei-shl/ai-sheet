7#[cfg(test)]
8mod content_types_tests {
9
10    use crate::content_types::ContentTypes;
11    use crate::test_functions::xml_to_vec;
12    use pretty_assertions::assert_eq;
13
14    #[test]
15    fn test_assemble() {
16        let mut content_types = ContentTypes::new();
17
18        content_types.add_default("jpeg", "image/jpeg");
19        content_types.add_worksheet_name(1);
20        content_types.add_share_strings();
21        content_types.add_override(
22            "/xl/workbook.xml",
23            "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet.main+xml",
24        );
25
26        content_types.assemble_xml_file();
27
28        let got = content_types.writer.read_to_str();
29        let got = xml_to_vec(got);
30
31        let expected = xml_to_vec(
32            r#"
33            <?xml version="1.0" encoding="UTF-8" standalone="yes"?>
34            <Types xmlns="http://schemas.openxmlformats.org/package/2006/content-types">
35              <Default Extension="rels" ContentType="application/vnd.openxmlformats-package.relationships+xml"/>
36              <Default Extension="xml" ContentType="application/xml"/>
37              <Default Extension="jpeg" ContentType="image/jpeg"/>
38              <Override PartName="/docProps/app.xml" ContentType="application/vnd.openxmlformats-officedocument.extended-properties+xml"/>
39              <Override PartName="/docProps/core.xml" ContentType="application/vnd.openxmlformats-package.core-properties+xml"/>
40              <Override PartName="/xl/styles.xml" ContentType="application/vnd.openxmlformats-officedocument.spreadsheetml.styles+xml"/>
41              <Override PartName="/xl/theme/theme1.xml" ContentType="application/vnd.openxmlformats-officedocument.theme+xml"/>
42              <Override PartName="/xl/worksheets/sheet1.xml" ContentType="application/vnd.openxmlformats-officedocument.spreadsheetml.worksheet+xml"/>
43              <Override PartName="/xl/sharedStrings.xml" ContentType="application/vnd.openxmlformats-officedocument.spreadsheetml.sharedStrings+xml"/>
44              <Override PartName="/xl/workbook.xml" ContentType="application/vnd.openxmlformats-officedocument.spreadsheetml.sheet.main+xml"/>
45             </Types>
46                "#,
47        );
48
49        assert_eq!(expected, got);
50    }
51}
