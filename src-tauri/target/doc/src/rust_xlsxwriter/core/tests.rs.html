7#[cfg(test)]
8mod core_tests {
9
10    use crate::core::Core;
11    use crate::ExcelDateTime;
12    use crate::{test_functions::xml_to_vec, DocProperties};
13
14    use pretty_assertions::assert_eq;
15
16    #[test]
17    fn test_assemble() {
18        let date = ExcelDateTime::from_ymd(2010, 1, 1).unwrap();
19        let properties = DocProperties::new()
20            .set_author("A User")
21            .set_creation_datetime(&date);
22
23        let mut core = Core::new();
24        core.properties = properties;
25
26        core.assemble_xml_file();
27
28        let got = core.writer.read_to_str();
29        let got = xml_to_vec(got);
30
31        let expected = xml_to_vec(
32            r#"
33            <?xml version="1.0" encoding="UTF-8" standalone="yes"?>
34            <cp:coreProperties xmlns:cp="http://schemas.openxmlformats.org/package/2006/metadata/core-properties" xmlns:dc="http://purl.org/dc/elements/1.1/" xmlns:dcterms="http://purl.org/dc/terms/" xmlns:dcmitype="http://purl.org/dc/dcmitype/" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance">
35              <dc:creator>A User</dc:creator>
36              <cp:lastModifiedBy>A User</cp:lastModifiedBy>
37              <dcterms:created xsi:type="dcterms:W3CDTF">2010-01-01T00:00:00Z</dcterms:created>
38              <dcterms:modified xsi:type="dcterms:W3CDTF">2010-01-01T00:00:00Z</dcterms:modified>
39            </cp:coreProperties>
40            "#,
41        );
42
43        assert_eq!(expected, got);
44    }
45}
