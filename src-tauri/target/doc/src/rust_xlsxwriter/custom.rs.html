9use crate::{xmlwriter::XMLWriter, CustomProperty, CustomPropertyType, DocProperties};
10
11pub struct Custom {
12    pub(crate) writer: XMLWriter,
13    pub(crate) properties: DocProperties,
14}
15
16impl Custom {
17    // -----------------------------------------------------------------------
18    // Public (and crate public) methods.
19    // -----------------------------------------------------------------------
20
21    // Create a new Custom struct.
22    pub(crate) fn new() -> Custom {
23        let writer = XMLWriter::new();
24
25        Custom {
26            writer,
27
28            properties: DocProperties::new(),
29        }
30    }
31
32    // -----------------------------------------------------------------------
33    // XML assembly methods.
34    // -----------------------------------------------------------------------
35
36    // Assemble and write the XML file.
37    pub(crate) fn assemble_xml_file(&mut self) {
38        self.writer.xml_declaration();
39
40        // Write the Properties element.
41        self.write_properties();
42
43        for (pid, property) in self.properties.custom_properties.clone().iter().enumerate() {
44            // Write the property element.
45            self.write_property(property, pid + 2);
46        }
47
48        // Close the final tag.
49        self.writer.xml_end_tag("Properties");
50    }
51
52    // Write the <Properties> element.
53    fn write_properties(&mut self) {
54        let schema = "http://schemas.openxmlformats.org/officeDocument/2006".to_string();
55        let xmlns = format!("{schema}/custom-properties");
56        let xmlns_vt = format!("{schema}/docPropsVTypes");
57
58        let attributes = [("xmlns", xmlns), ("xmlns:vt", xmlns_vt)];
59
60        self.writer.xml_start_tag("Properties", &attributes);
61    }
62
63    // Write the <property> element.
64    fn write_property(&mut self, property: &CustomProperty, pid: usize) {
65        let fmtid = "{D5CDD505-2E9C-101B-9397-08002B2CF9AE}".to_string();
66        let attributes = [
67            ("fmtid", fmtid),
68            ("pid", pid.to_string()),
69            ("name", property.name.to_string()),
70        ];
71
72        self.writer.xml_start_tag("property", &attributes);
73
74        match property.property_type {
75            CustomPropertyType::Int => self.write_vt_i_4(property.number_int),
76            CustomPropertyType::Bool => self.write_vt_bool(property.boolean),
77            CustomPropertyType::Real => self.write_vt_r_8(property.number_real),
78            CustomPropertyType::Text => self.write_vt_lpwstr(&property.text),
79            CustomPropertyType::DateTime => self.write_vt_filetime(&property.datetime),
80        }
81
82        self.writer.xml_end_tag("property");
83    }
84
85    // Write the <vt:lpwstr> element.
86    fn write_vt_lpwstr(&mut self, text: &str) {
87        self.writer.xml_data_element_only("vt:lpwstr", text);
88    }
89
90    // Write the <vt:filetime> element.
91    fn write_vt_filetime(&mut self, utc_datetime: &str) {
92        self.writer
93            .xml_data_element_only("vt:filetime", utc_datetime);
94    }
95
96    // Write the <vt:i4> element.
97    fn write_vt_i_4(&mut self, number: i32) {
98        self.writer
99            .xml_data_element_only("vt:i4", &number.to_string());
100    }
101
102    // Write the <vt:r8> element.
103    fn write_vt_r_8(&mut self, number: f64) {
104        self.writer
105            .xml_data_element_only("vt:r8", &number.to_string());
106    }
107
108    // Write the <vt:bool> element.
109    fn write_vt_bool(&mut self, boolean: bool) {
110        self.writer
111            .xml_data_element_only("vt:bool", &boolean.to_string());
112    }
113}
