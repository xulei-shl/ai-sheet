7#[cfg(test)]
8mod custom_tests {
9
10    use crate::custom::Custom;
11    use crate::ExcelDateTime;
12    use crate::{test_functions::xml_to_vec, DocProperties};
13    use pretty_assertions::assert_eq;
14
15    #[test]
16    fn test_assemble1() {
17        let mut custom = Custom::new();
18
19        let properties = DocProperties::new().set_custom_property("Checked by", "Adam");
20
21        custom.properties = properties;
22
23        custom.assemble_xml_file();
24
25        let got = custom.writer.read_to_str();
26        let got = xml_to_vec(got);
27
28        let expected = xml_to_vec(
29            r#"
30            <?xml version="1.0" encoding="UTF-8" standalone="yes"?>
31            <Properties xmlns="http://schemas.openxmlformats.org/officeDocument/2006/custom-properties" xmlns:vt="http://schemas.openxmlformats.org/officeDocument/2006/docPropsVTypes">
32              <property fmtid="{D5CDD505-2E9C-101B-9397-08002B2CF9AE}" pid="2" name="Checked by">
33                <vt:lpwstr>Adam</vt:lpwstr>
34              </property>
35            </Properties>
36            "#,
37        );
38
39        assert_eq!(expected, got);
40    }
41
42    #[test]
43    fn test_assemble2() {
44        let date = ExcelDateTime::from_ymd(2016, 12, 12)
45            .unwrap()
46            .and_hms(23, 0, 0)
47            .unwrap();
48
49        let mut custom = Custom::new();
50
51        let properties = DocProperties::new()
52            .set_custom_property("Checked by", "Adam")
53            .set_custom_property("Date completed", &date)
54            .set_custom_property("Document number", 12345)
55            .set_custom_property("Reference", 1.2345)
56            .set_custom_property("Source", true)
57            .set_custom_property("Status", false)
58            .set_custom_property("Department", "Finance")
59            .set_custom_property("Group", 1.2345678901234);
60
61        custom.properties = properties;
62
63        custom.assemble_xml_file();
64
65        let got = custom.writer.read_to_str();
66        let got = xml_to_vec(got);
67
68        let expected = xml_to_vec(
69            r#"
70            <?xml version="1.0" encoding="UTF-8" standalone="yes"?>
71            <Properties xmlns="http://schemas.openxmlformats.org/officeDocument/2006/custom-properties" xmlns:vt="http://schemas.openxmlformats.org/officeDocument/2006/docPropsVTypes">
72              <property fmtid="{D5CDD505-2E9C-101B-9397-08002B2CF9AE}" pid="2" name="Checked by">
73                <vt:lpwstr>Adam</vt:lpwstr>
74              </property>
75              <property fmtid="{D5CDD505-2E9C-101B-9397-08002B2CF9AE}" pid="3" name="Date completed">
76                <vt:filetime>2016-12-12T23:00:00Z</vt:filetime>
77              </property>
78              <property fmtid="{D5CDD505-2E9C-101B-9397-08002B2CF9AE}" pid="4" name="Document number">
79                <vt:i4>12345</vt:i4>
80              </property>
81              <property fmtid="{D5CDD505-2E9C-101B-9397-08002B2CF9AE}" pid="5" name="Reference">
82                <vt:r8>1.2345</vt:r8>
83              </property>
84              <property fmtid="{D5CDD505-2E9C-101B-9397-08002B2CF9AE}" pid="6" name="Source">
85                <vt:bool>true</vt:bool>
86              </property>
87              <property fmtid="{D5CDD505-2E9C-101B-9397-08002B2CF9AE}" pid="7" name="Status">
88                <vt:bool>false</vt:bool>
89              </property>
90              <property fmtid="{D5CDD505-2E9C-101B-9397-08002B2CF9AE}" pid="8" name="Department">
91                <vt:lpwstr>Finance</vt:lpwstr>
92              </property>
93              <property fmtid="{D5CDD505-2E9C-101B-9397-08002B2CF9AE}" pid="9" name="Group">
94                <vt:r8>1.2345678901234</vt:r8>
95              </property>
96            </Properties>
97            "#,
98        );
99
100        assert_eq!(expected, got);
101    }
