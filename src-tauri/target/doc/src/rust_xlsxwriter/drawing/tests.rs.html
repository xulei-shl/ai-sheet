7#[cfg(test)]
8mod drawing_tests {
9
10    use crate::drawing::*;
11    use crate::test_functions::xml_to_vec;
12    use pretty_assertions::assert_eq;
13
14    #[test]
15    fn test_assemble() {
16        let mut drawing = Drawing::new();
17
18        let from = DrawingCoordinates {
19            col: 2,
20            row: 1,
21            col_offset: 0.0,
22            row_offset: 0.0,
23        };
24
25        let to = DrawingCoordinates {
26            col: 3,
27            row: 6,
28            col_offset: 533257.0,
29            row_offset: 190357.0,
30        };
31
32        let drawing_info = DrawingInfo {
33            from,
34            to,
35            col_absolute: 1219200,
36            row_absolute: 190500,
37            width: 1142857.0,
38            height: 1142857.0,
39            name: "Picture 1".to_string(),
40            description: "rust.png".to_string(),
41            decorative: false,
42            rel_id: 1,
43            object_movement: ObjectMovement::MoveButDontSizeWithCells,
44            drawing_type: DrawingType::Image,
45            url: None,
46            is_portrait: false,
47        };
48
49        drawing.drawings.push(drawing_info);
50
51        drawing.assemble_xml_file();
52
53        let got = drawing.writer.read_to_str();
54        let got = xml_to_vec(got);
55
56        let expected = xml_to_vec(
57            r#"
58                <?xml version="1.0" encoding="UTF-8" standalone="yes"?>
59                <xdr:wsDr xmlns:xdr="http://schemas.openxmlformats.org/drawingml/2006/spreadsheetDrawing" xmlns:a="http://schemas.openxmlformats.org/drawingml/2006/main">
60                <xdr:twoCellAnchor editAs="oneCell">
61                    <xdr:from>
62                    <xdr:col>2</xdr:col>
63                    <xdr:colOff>0</xdr:colOff>
64                    <xdr:row>1</xdr:row>
65                    <xdr:rowOff>0</xdr:rowOff>
66                    </xdr:from>
67                    <xdr:to>
68                    <xdr:col>3</xdr:col>
69                    <xdr:colOff>533257</xdr:colOff>
70                    <xdr:row>6</xdr:row>
71                    <xdr:rowOff>190357</xdr:rowOff>
72                    </xdr:to>
73                    <xdr:pic>
74                    <xdr:nvPicPr>
75                        <xdr:cNvPr id="2" name="Picture 1" descr="rust.png"/>
76                        <xdr:cNvPicPr>
77                        <a:picLocks noChangeAspect="1"/>
78                        </xdr:cNvPicPr>
79                    </xdr:nvPicPr>
80                    <xdr:blipFill>
81                        <a:blip xmlns:r="http://schemas.openxmlformats.org/officeDocument/2006/relationships" r:embed="rId1"/>
82                        <a:stretch>
83                        <a:fillRect/>
84                        </a:stretch>
85                    </xdr:blipFill>
86                    <xdr:spPr>
87                        <a:xfrm>
88                        <a:off x="1219200" y="190500"/>
89                        <a:ext cx="1142857" cy="1142857"/>
90                        </a:xfrm>
91                        <a:prstGeom prst="rect">
92                        <a:avLst/>
93                        </a:prstGeom>
94                    </xdr:spPr>
95                    </xdr:pic>
96                    <xdr:clientData/>
97                </xdr:twoCellAnchor>
98                </xdr:wsDr>
99                "#,
100        );
101
102        assert_eq!(expected, got);
103    }
104}
