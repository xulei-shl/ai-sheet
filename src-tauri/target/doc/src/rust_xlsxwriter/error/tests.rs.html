7#[cfg(test)]
8mod error_tests {
9
10    use std::io::{Error, ErrorKind};
11
12    use crate::XlsxError;
13    use pretty_assertions::assert_eq;
14    use zip::result::ZipError;
15
16    #[test]
17    fn test_error_display() {
18        let name = "ERROR";
19
20        assert_eq!(
21            XlsxError::RowColumnLimitError.to_string(),
22            "Row or column exceeds Excel's allowed limits (1,048,576 x 16,384)."
23        );
24        assert_eq!(
25            XlsxError::RowColumnOrderError.to_string(),
26            "First row or column in range is greater than last row or column."
27        );
28        assert_eq!(
29            XlsxError::SheetnameCannotBeBlank(name.to_string()).to_string(),
30            "Worksheet name 'ERROR' cannot be blank."
31        );
32        assert_eq!(
33            XlsxError::SheetnameLengthExceeded(name.to_string()).to_string(),
34            "Worksheet name 'ERROR' exceeds Excel's limit of 31 characters."
35        );
36        assert_eq!(
37            XlsxError::SheetnameReused(name.to_string()).to_string(),
38            "Worksheet name 'ERROR' has already been used in this workbook."
39        );
40        assert_eq!(
41            XlsxError::SheetnameContainsInvalidCharacter(name.to_string()).to_string(),
42            "Worksheet name 'ERROR' cannot contain invalid characters: '[ ] : * ? / \\'."
43        );
44        assert_eq!(
45            XlsxError::SheetnameStartsOrEndsWithApostrophe(name.to_string()).to_string(),
46            "Worksheet name 'ERROR' cannot start or end with an apostrophe."
47        );
48        assert_eq!(
49            XlsxError::MaxStringLengthExceeded.to_string(),
50            "String exceeds Excel's limit of 32,767 characters."
51        );
52        assert_eq!(
53            XlsxError::UnknownWorksheetNameOrIndex(name.to_string()).to_string(),
54            "Unknown Worksheet name or index 'ERROR'."
55        );
56        assert_eq!(
57            XlsxError::MergeRangeSingleCell.to_string(),
58            "A merge range cannot be a single cell in Excel."
59        );
60        assert_eq!(
61            XlsxError::MergeRangeOverlaps(name.to_string(), name.to_string()).to_string(),
62            "Merge range ERROR overlaps with previous merge range ERROR."
63        );
64
65        assert_eq!(
66            XlsxError::IoError(Error::new(ErrorKind::Other, "ERROR")).to_string(),
67            "ERROR"
68        );
69        assert_eq!(
70            XlsxError::ZipError(ZipError::FileNotFound).to_string(),
71            "specified file not found in archive"
72        );
73
74        let result = catch_zip_error();
75        assert!(matches!(result, Err(XlsxError::ZipError(_))));
76
77        let result = catch_io_error();
78        assert!(matches!(result, Err(XlsxError::IoError(_))));
79
80        assert_eq!(
81            format!("{:?}", XlsxError::RowColumnLimitError),
82            "RowColumnLimitError"
83        );
84    }
85
86    fn catch_zip_error() -> Result<(), XlsxError> {
87        throw_zip_error()?;
88        Ok(())
89    }
90
91    fn throw_zip_error() -> Result<(), ZipError> {
92        Err(ZipError::FileNotFound)
93    }
94
95    fn catch_io_error() -> Result<(), XlsxError> {
96        throw_io_error()?;
97        Ok(())
98    }
99
100    fn throw_io_error() -> Result<(), std::io::Error> {
101        Err(Error::new(ErrorKind::Other, "ERROR"))
102    }
103}
