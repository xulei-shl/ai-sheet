243#[test]
244fn test_parse_function_escapes() {
245    use crate::Formula;
246    // Slightly more rigorous tests of function edge case.
247    let formula_strings = vec![
248        // Test simple escapes for formulas.
249        ("", "", false),
250        ("FOO()", "FOO()", false),
251        ("=FOO()", "FOO()", false),
252        ("{=FOO()}", "FOO()", false),
253        ("FOO() + A1", "FOO() + A1", false),
254        (r#""FOO()""#, r#""FOO()""#, false),
255        (r#""FOO() 😀""#, r#""FOO() 😀""#, false),
256        (r#" "😀" & FOO()"#, r#" "😀" & FOO()"#, false),
257        (r#""FOO()"&"BAR()""#, r#""FOO()"&"BAR()""#, false),
258        (r#"""""&FOO()"#, r#"""""&FOO()"#, false),
259        ("SEQUENCE(10)", "_xlfn.SEQUENCE(10)", true),
260        (
261            r#"=INDEX(C:C,MATCH(MINIFS(A:A,A:A,">="&EDATE(A5,1)),A:A))"#,
262            r#"INDEX(C:C,MATCH(_xlfn.MINIFS(A:A,A:A,">="&EDATE(A5,1)),A:A))"#,
263            false,
264        ),
265        ("POISSON.DIST(A1:A3)", "_xlfn.POISSON.DIST(A1:A3)", false),
266        (
267            "SEQUENCE(10) + POISSON.DIST(A1:A3)",
268            "_xlfn.SEQUENCE(10) + _xlfn.POISSON.DIST(A1:A3)",
269            true,
270        ),
271        (
272            "_xlfn.SEQUENCE(10) + _xlfn.POISSON.DIST(A1:A3)",
273            "_xlfn.SEQUENCE(10) + _xlfn.POISSON.DIST(A1:A3)",
274            true,
275        ),
276        ("FILTER(A1:A10,1)", "_xlfn._xlws.FILTER(A1:A10,1)", true),
277    ];
278
279    for &(input_string, expected_formula, expected_dynamic) in &formula_strings {
280        let formula = Formula::new(input_string);
281        assert_eq!(formula.formula_string, expected_formula);
282        assert_eq!(formula.has_dynamic_function, expected_dynamic);
283    }
284}

286#[test]
287fn test_escape_table_functions() {
288    use crate::Formula;
289    // Test table escapes.
290    let formula_strings = vec![
291        ("", ""),
292        ("@", "[#This Row],"),
293        (r#""@""#, r#""@""#),
294        (
295            "SUM(Table1[@[Column1]:[Column3]])",
296            "SUM(Table1[[#This Row],[Column1]:[Column3]])",
297        ),
298        (
299            r#"=HYPERLINK(CONCAT("http://myweb.com:1677/'@md=d&path/to/sour/...'@/",[@CL],"?ac=10"),[@CL])"#,
300            r#"HYPERLINK(_xlfn.CONCAT("http://myweb.com:1677/'@md=d&path/to/sour/...'@/",[[#This Row],CL],"?ac=10"),[[#This Row],CL])"#,
301        ),
302    ];
303
304    for &(input_string, expected_formula) in &formula_strings {
305        let formula = Formula::new(input_string).escape_table_functions();
306
307        assert_eq!(formula.formula_string, expected_formula);
308    }
